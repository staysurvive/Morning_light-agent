interface AnalyticsOverview {

  totalTokens: number

  date: string
  calls: number
  inputTokens: number
  outputTokens: number

interface AgentStats {

}

interface ModelStats {

### 评估数据结构

```typescript
interface EvaluationOverview {
  avgSatisfaction: number         // 平均满意度百分比
  annotationCoverage: number      // 标注覆盖率
  needsOptimization: number       // 需优化数量
  avgRating: number               // 平均评分(1-5)
}

interface AgentEvaluation {
  agentId: string
  agentName: string
  satisfaction: number
  avgRating: number
  annotationCount: number
  needsOptimization: number
  trend: 'up' | 'down' | 'stable'
}

interface CommonIssue {
  type: string
  count: number
  percentage: number
  trend: 'up' | 'down' | 'stable'
}
```

## Mock数据要点

### 用量统计
- 30天的每日统计数据
- 按Agent和模型的分布数据
- 数据有合理的工作日/周末波动

### 费用分析
- 30天的每日费用数据
- 按模型和Agent的费用分布
- 预算使用进度

### 效果评估
- 每个Agent的评估数据
- 常见问题分类统计
- 满意度趋势数据
